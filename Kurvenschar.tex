    % Plot-Achsen deklarieren
    \psaxes[
        tickcolor=HTWGreen100

    % Plotten einer Kurvenschar eines Polynoms 5. Grades (4 Funktionen)

        \psplot[
            linecolor=red!\nB,
            showpoints,
            dotsize=0.1
        ]{-5}{5}{%
            ((\nA*x+4)*(x+1)*(x-1)*(x-3)*(x-5))/100} % gew√§hltes Polynom
